[JS] select 에 option 설정 새로 하기

select 객체에 있는 기존 option 객체를 모두 삭제하고, 새로운 option 객체들로 채워넣기.

var selectObject = document.getElementsByName("selectObjectName")[0];

// 모든 Option 들 삭제
while (selectObject.length != 0) {
selectObject.options.remove(0);
}

// 새로운 Option 추가
var option = new Option("출력문자열", "값");
selectObject.add(option);
// 계속 새로운 Option 객체를 만들어 추가한다.


참조 : option 객체
