[UNIX-SH] 파일 확장자를 일괄적으로 바꾸자

파일의 확장자를 일괄적으로 바꿔준다.

대문자로 된 확장자를 소문자로 하거나 하기위해 만든 것~


#!/bin/sh
# chext - 파일의 확장자를 일괄적으로 바꿔주는 프로그램
# http://myhome.naver.com/innocens

# chext [-nm] 확장자1 확장자2
#    : 확장자1을 가진 파일들을 모두 확장자2를 갖도록 고친다.
#    : -nm 메시지를 출력하지 않음

# howToUse() : 프로그램 사용법을 출력한다.
howToUse() {
    echo "Usage : $0 [-nm] ext1 ext2"
    echo "        -nm : Don't print messages"
}

# 프로그램의 시작
message=on

if [ "$1" = "-nm" ]; then
    if [ $# = "3" ]; then
ext1=$2
ext2=$3
message=off
    else
howToUse
exit 1
    fi
elif [ "$#" = "2" ]; then
    ext1=$1
    ext2=$2
else
    howToUse
    exit 1
fi

# 이유는 모르겠지만 아래의 for 문에서 $ext1의 확장자를 가진 파일이
# 존재하지 않을 경우 $file 변수에는 *.$ext1 이 대입된다.

for file in *.$ext1; do
    if [ $file = "*.$ext1" ]; then
echo "$ext1 의 확장자를 가진 파일이 존재하지 않습니다."
exit 1
    fi

    if [ $message = "on" ]; then
echo "$file --> ${file%$ext1}$ext2"
    fi
    mv $file ${file%$ext1}$ext2
done

exit 0
